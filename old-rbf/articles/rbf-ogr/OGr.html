def rbf(r):
    return np.exp(-eps * r**2)
def rbfd(r):
    return -2*eps*r*rbf(r)
def rbfd_div_r(r):
    return -2*eps*rbf(r)

def rbfd2(r):
    return 2*eps*rbf(r)*(2*eps*r**2-1)
def rbfd3(r):
    return 4*eps**2*r*(-2*eps*r**2 + 3)*rbf(r)

def rbfd3_div_r(r):
    return 4*eps**2*(-2*eps*r**2 + 3)*rbf(r)

def rbfd2_div_r2(r):
    if r== 0:
        return 0
    return 2*eps*rbf(r)*(2*eps*r**2-1)/r**2


def rbfL(r):
    return rbfd2(r) + rbfd_div_r(r) *2

def drbfL_div_r(r):
    #return 1/(r+MEPS)**3 * rbfd(r) - 1/(r+MEPS)**2 * rbfd2(r) - 1/(r+MEPS) * rbfd3(r)
    return 1/(r) * rbfd3(r)

def grad_rbf(node, center):
    r = dist(node, center)
    return rbfd_div_r(r) * (node - center)

def rbf_x(node, center, xi):
    x = (node-center)[xi]
    r = dist(node, center)
    return rbfd_div_r(r)*x

def rbf_xx(node, center, xi):
    x = (node-center)[xi]
    r = dist(node, center)
    if r==0:
        return -2
    return rbfd_div_r(r)* (1 - (x/r)**2) + (x/r)**2 * rbfd2(r)

def rbf_xy(node, center, xi, yi):
    x = (node-center)[xi]
    y = (node-center)[yi]
    r = dist(node, center)
    if r == 0:
        return 0
    return (x*y/r**2) * (rbfd2(r) - rbfd_div_r(r))

def rbf(r):
    return np.exp(-eps * r**2)

def zeta(r): # 1/r * d/dr rbf 
    return -2*eps*rbf(r)

def chi(r): # 1/r * d/dr zeta 
    return 4*eps**2 *rbf(r)


def beta(r): # 1/r * d/dr chi 
    return -8 * eps**3 * rbf(r)

def rbfL(r):
    return 3*zeta(r) + r**2 * chi(r)
