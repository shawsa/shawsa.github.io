def toep_mult(points, x):
    N = len(x)
    #a = np.block([A[:,0], A[0,:0:-1]] )
    a = np.block([points, points[:0:-1]])
    a_hat = fft.fft(a)
    x_hat = fft.fft(np.block([x, np.zeros(N-1)]))
    b_hat = a_hat * x_hat # * len(a_hat) - Scipy's FFT isn't scaled
    b = fft.ifft(b_hat)
    return b[:N]

def toep_inv(points, b):
